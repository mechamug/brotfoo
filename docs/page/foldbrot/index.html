            y += zy*count/p;
            n--;
            p++;
        }
        var new_zx = zx*zx - zy*zy + x;
        zy = 2*zx*zy + y;
        zx = new_zx;
        count++;
    }
    var angle = Math.floor((Math.atan(zy/zx)+Math.PI/2)/Math.PI*count);
    return count < maxIterations ? count : angle;
